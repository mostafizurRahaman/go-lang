Goroutine-দের মধ্যে যোগাযোগের মূল হাতিয়ার — Channel। এই গাইডে প্রতিটি ধরনের Channel, সব রকমের প্যাটার্ন এবং বাস্তব উদাহরণ দিয়ে শিখুন সম্পূর্ণ বাংলায়।

01

Channel কী এবং কেন দরকার?

Go প্রোগ্রামিং ভাষায় Channel হল দুটি Goroutine-এর মধ্যে ডেটা পাঠানোর একটি পাইপ বা নালী। একটি Goroutine Channel-এ ডেটা পাঠায়, আরেকটি সেটি গ্রহণ করে।

Go-এর বিখ্যাত দর্শন হল:

💬 Go Philosophy

"Do not communicate by sharing memory; instead, share memory by communicating."
অর্থাৎ — শেয়ার্ড মেমরি দিয়ে যোগাযোগ করো না, বরং যোগাযোগের মাধ্যমে মেমরি শেয়ার করো।

Channel ছাড়া কী সমস্যা?

⚠️ Race Condition

একাধিক Goroutine একই ভেরিয়েবল পড়লে/লিখলে ডেটা নষ্ট হয়। Channel এটি এড়ায়।

🔒 Mutex জটিলতা

Mutex দিয়ে lock/unlock ম্যানেজ করা কঠিন ও বাগ-প্রবণ। Channel বেশি পরিষ্কার।

📡 সমন্বয়হীনতা

কোন Goroutine কখন শেষ হল জানা কঠিন। Channel দিয়ে সহজে সংকেত দেওয়া যায়।

💥 Deadlock

ভুল ব্যবহারে Deadlock হয়। Channel-এর নিয়ম বোঝা থাকলে এড়ানো সহজ।

02

Channel-এর ধরনসমূহ

ধরন সংজ্ঞা Buffer আছে? Block করে কখন?
make(chan T) Unbuffered Channel না সবসময় — যতক্ষণ না অন্যপাশ প্রস্তুত
make(chan T, n) Buffered Channel হ্যাঁ (n টি) Buffer পূর্ণ হলে send; খালি হলে receive
chan<- T Send-only Channel উভয় উপরের মতো, শুধু পাঠানো যায়
<-chan T Receive-only Channel উভয় উপরের মতো, শুধু গ্রহণ করা যায়
var ch chan T Nil Channel না চিরকাল block করে
03

Channel তৈরি ও মৌলিক অপারেশন

Channel তৈরি করতে make() ফাংশন ব্যবহার করতে হয়। তিনটি মৌলিক অপারেশন আছে:

go channel-basic.go 
package main

import "fmt"

func main() {
    // ১. Channel তৈরি করা
    ch := make(chan int)  // Unbuffered channel

    // ২. Goroutine-এ পাঠানো (অবশ্যই goroutine-এ, নয়তো deadlock)
    go func() {
        ch <- 42  // Send: channel-এ 42 পাঠাও
    }()

    // ৩. গ্রহণ করা
    value := <-ch  // Receive: channel থেকে পাও
    fmt.Println("পাওয়া গেছে:", value)  // পাওয়া গেছে: 42

    // ৪. গ্রহণ করে ফেলে দেওয়া (শুধু sync করতে)
    done := make(chan struct{})
    go func() {
        fmt.Println("কাজ হচ্ছে...")
        done <- struct{}{}  // সংকেত পাঠাও
    }()
    <-done  // সংকেতের জন্য অপেক্ষা করো
    fmt.Println("কাজ শেষ!")
}
ℹ️ অপারেটর

<-ch মানে "channel থেকে পাও" · ch <- v মানে "channel-এ v পাঠাও"। তীরের দিক লক্ষ্য করো — সবসময় channel-এর দিকে যায়।

04

Unbuffered Channel

Unbuffered Channel-এ কোনো জায়গা নেই। প্রেরক পাঠাতে পারবে না যতক্ষণ না গ্রাহক প্রস্তুত থাকে, এবং গ্রাহক পাবে না যতক্ষণ না প্রেরক পাঠায়। এটি synchronous handoff — হাতে হাত বদল।

Sender
ch ← 42
🔴 block
Channel
(ফাঁকা)
🔴 block
Receiver
← ch

উভয়পক্ষ প্রস্তুত হলেই ডেটা বদল হয় — একসাথে!

go unbuffered-channel.go 
package main

import (
    "fmt"
    "time"
)

func main() {
    ch := make(chan string)  // Unbuffered

    // প্রেরক goroutine
    go func() {
        fmt.Println("[প্রেরক] পাঠানোর আগে...")
        time.Sleep(1 * time.Second)
        ch <- "নমস্কার!"  // গ্রাহক প্রস্তুত না হওয়া পর্যন্ত block
        fmt.Println("[প্রেরক] পাঠানো হয়েছে!")
    }()

    fmt.Println("[গ্রাহক] প্রাপ্তির অপেক্ষায়...")
    msg := <-ch  // প্রেরক না পাঠানো পর্যন্ত block
    fmt.Println("[গ্রাহক] পেয়েছি:", msg)
}

/* আউটপুট:
[গ্রাহক] প্রাপ্তির অপেক্ষায়...
[প্রেরক] পাঠানোর আগে...
[প্রেরক] পাঠানো হয়েছে!
[গ্রাহক] পেয়েছি: নমস্কার!
*/

Unbuffered-এর ব্যবহারক্ষেত্র

  • Synchronization: দুটি Goroutine-এর কাজ একটি নির্দিষ্ট পয়েন্টে মেলানো
  • Guarantee: নিশ্চিত করা যে ডেটা পাঠানোর আগেই গ্রাহক প্রস্তুত
  • Ping-Pong: পর্যায়ক্রমিক হাতবদল প্যাটার্ন
05

Buffered Channel

Buffered Channel-এ নির্দিষ্ট সংখ্যক আইটেম রাখার জায়গা আছে। Buffer পূর্ণ না হওয়া পর্যন্ত প্রেরক block হয় না। এটি asynchronous।

Sender
→
[ 1 | 2 | 3 ]
Buffer (cap=3)
→
Receiver
go buffered-channel.go 
package main

import "fmt"

func main() {
    // ধারণক্ষমতা ৩-এর buffered channel
    ch := make(chan int, 3)

    // গ্রাহক ছাড়াই তিনটি পাঠানো যাবে — block হবে না
    ch <- 10
    ch <- 20
    ch <- 30
    // ch <- 40  // এটি লিখলে deadlock! buffer পূর্ণ

    fmt.Println("Buffer-এ আছে:", len(ch), "টি")       // 3
    fmt.Println("মোট ধারণক্ষমতা:", cap(ch), "টি")  // 3

    // পড়া
    fmt.Println(<-ch)  // 10
    fmt.Println(<-ch)  // 20
    fmt.Println(<-ch)  // 30
}

// ─── Unbuffered vs Buffered তুলনা ───
func comparison() {
    // Unbuffered: পাঠাতে গেলেই block, গ্রাহক না থাকলে
    ub := make(chan int)

    // Buffered: buffer খালি থাকলে পাঠানো যায়
    bf := make(chan int, 5)

    _ = ub
    _ = bf
}
⚠️ সতর্কতা

Buffered Channel-এর আকার ঠিক না করলে সমস্যা হয়। অনেক বড় buffer মেমরি নষ্ট করে; অনেক ছোট buffer Deadlock ডাকে।

len() এবং cap() ফাংশন

ফাংশনকী করেউদাহরণ
len(ch)এখন buffer-এ কতটি আইটেম আছেlen(ch) == 2
cap(ch)মোট কতটি আইটেম ধরেcap(ch) == 5
06

Directional (দিকনির্দেশক) Channel

ফাংশনের প্যারামিটারে Channel-এর দিক নির্দিষ্ট করা যায়। এতে কোড নিরাপদ হয় — ভুল দিকে পাঠালে compile-time error পাওয়া যায়।

go directional-channel.go 
package main

import "fmt"

// শুধু পাঠাতে পারে: chan<- int
func producer(out chan<- int) {
    for i := 1; i <= 5; i++ {
        out <- i  // ✅ পাঠানো যাবে
    }
    close(out)
    // <-out  // ❌ compile error: cannot receive from send-only channel
}

// শুধু গ্রহণ করতে পারে: <-chan int
func consumer(in <-chan int) {
    for v := range in {
        fmt.Println("পেয়েছি:", v)
    }
    // in <- 5  // ❌ compile error: cannot send to receive-only channel
}

func main() {
    ch := make(chan int, 5)

    // bidirectional channel কে directional হিসেবে পাস করা যায়
    go producer(ch)   // chan int → chan<- int (স্বয়ংক্রিয় রূপান্তর)
    consumer(ch)     // chan int → <-chan int (স্বয়ংক্রিয় রূপান্তর)
}

/* আউটপুট:
পেয়েছি: 1
পেয়েছি: 2
পেয়েছি: 3
পেয়েছি: 4
পেয়েছি: 5
*/
💡 সেরা অভ্যাস

ফাংশনে Channel পাস করার সময় সবসময় দিক নির্দিষ্ট করো। এটি কোডের উদ্দেশ্য স্পষ্ট করে এবং বাগ কমায়।

07

Range দিয়ে Channel পড়া

for range দিয়ে Channel থেকে পর্যায়ক্রমে সব ডেটা পড়া যায়। Channel বন্ধ (close()) না হওয়া পর্যন্ত loop চলতে থাকে।

go range-channel.go 
package main

import "fmt"

func generate(nums ...int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)  // কাজ শেষে অবশ্যই বন্ধ করো
        for _, n := range nums {
            out <- n
        }
    }()
    return out
}

func main() {
    ch := generate(10, 20, 30, 40, 50)

    // range channel — close() না হওয়া পর্যন্ত চলবে
    for v := range ch {
        fmt.Println(v)
    }
    fmt.Println("শেষ!")

    // ─── ok idiom: channel বন্ধ কিনা পরীক্ষা ───
    ch2 := generate(1, 2)
    v1, ok1 := <-ch2
    fmt.Println(v1, ok1)  // 1, true  (channel খোলা)
    v2, ok2 := <-ch2
    fmt.Println(v2, ok2)  // 2, true
    v3, ok3 := <-ch2
    fmt.Println(v3, ok3)  // 0, false (channel বন্ধ, zero value)
}
08

Channel বন্ধ করা

close(ch) দিয়ে channel বন্ধ করা হয়। এরপর কেউ পাঠাতে পারবে না, কিন্তু বাকি ডেটা পড়া যাবে।

go close-channel.go 
package main

import "fmt"

func main() {
    ch := make(chan int, 3)
    ch <- 1
    ch <- 2
    ch <- 3
    close(ch)  // channel বন্ধ

    // বন্ধ channel থেকে বাকি ডেটা পড়া যায়
    for v := range ch {
        fmt.Println(v)  // 1, 2, 3
    }

    // বন্ধ channel থেকে পড়লে zero value আসে
    v, ok := <-ch
    fmt.Println(v, ok)  // 0, false

    // close() শুধু প্রেরকই করবে, গ্রাহক কখনো নয়!
    // defer close(ch) ব্যবহার করো নিরাপদ থাকতে
}
🚨 বন্ধ channel-এ পাঠানো Panic

বন্ধ channel-এ পাঠালে প্যানিক হয়: send on closed channel। সবসময় শুধু প্রেরকই channel বন্ধ করবে। গ্রাহক কখনো close() ডাকবে না।

পরিস্থিতিফলাফল
খোলা channel-এ send (গ্রাহক নেই)Block হয়
বন্ধ channel-এ sendPanic!
বন্ধ channel থেকে receive (ডেটা আছে)ডেটা পায়, ok=true
বন্ধ channel থেকে receive (ডেটা নেই)zero value পায়, ok=false
nil channel-এ send বা receiveচিরকাল block
09

Select Statement উন্নত

select হল Channel-এর জন্য switch। একাধিক channel-এর মধ্যে যেটি প্রস্তুত সেটি থেকে কাজ করে। যদি একাধিক প্রস্তুত থাকে, র্যান্ডমলি একটি বেছে নেয়।

go select-statement.go 
package main

import (
    "fmt"
    "time"
)

func main() {
    ch1 := make(chan string)
    ch2 := make(chan string)

    go func() {
        time.Sleep(2 * time.Second)
        ch1 <- "একটি"
    }()

    go func() {
        time.Sleep(1 * time.Second)
        ch2 <- "দুই"
    }()

    // যে channel আগে প্রস্তুত সেটি থেকে নেবে
    for i := 0; i < 2; i++ {
        select {
        case msg1 := <-ch1:
            fmt.Println("ch1 থেকে:", msg1)
        case msg2 := <-ch2:
            fmt.Println("ch2 থেকে:", msg2)
        }
    }

    // ─── Default case: non-blocking ───
    ch3 := make(chan int)
    select {
    case v := <-ch3:
        fmt.Println("পেলাম:", v)
    default:
        fmt.Println("channel প্রস্তুত নেই, এগিয়ে যাচ্ছি")
    }
}

/* আউটপুট:
ch2 থেকে: দুই    (১ সেকেন্ড পরে)
ch1 থেকে: একটি  (২ সেকেন্ড পরে)
channel প্রস্তুত নেই, এগিয়ে যাচ্ছি
*/

Select-এর গুরুত্বপূর্ণ বৈশিষ্ট্য

  • একাধিক case প্রস্তুত থাকলে র্যান্ডমলি একটি বেছে নেয়
  • default case থাকলে কোনো channel প্রস্তুত না থাকলেও block হয় না
  • select {} (খালি) চিরকাল block করে — ইচ্ছাকৃত block করতে ব্যবহার হয়
  • nil channel-এর case কখনো নির্বাচিত হয় না
10

Timeout প্যাটার্ন

time.After() ব্যবহার করে select-এর সাথে Timeout বাস্তবায়ন করা যায়। এটি দীর্ঘ অপেক্ষা এড়াতে ব্যবহৃত হয়।

go timeout-pattern.go 
package main

import (
    "fmt"
    "time"
)

func slowOperation() <-chan string {
    result := make(chan string)
    go func() {
        time.Sleep(3 * time.Second)  // ধীর কাজ
        result <- "কাজ শেষ!"
    }()
    return result
}

func main() {
    result := slowOperation()

    select {
    case r := <-result:
        fmt.Println("সফল:", r)

    case <-time.After(2 * time.Second):
        fmt.Println("সময় শেষ! অপারেশন timeout হয়েছে")
    }

    // ─── Context দিয়ে আরো ভালো timeout ───
    // ctx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
    // defer cancel()
    // select { case r := <-result: ... case <-ctx.Done(): ... }
}

/* আউটপুট (২ সেকেন্ড পরে):
সময় শেষ! অপারেশন timeout হয়েছে
*/

package main

import "fmt"

// Stage 1: সংখ্যা তৈরি করো
func generate(nums ...int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for _, n := range nums {
            out <- n
        }
    }()
    return out
}

// Stage 2: বর্গ করো
func square(in <-chan int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for n := range in {
            out <- n * n
        }
    }()
    return out
}

// Stage 3: ফিল্টার করো (শুধু বড় মান)
func filter(in <-chan int, threshold int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for n := range in {
            if n > threshold {
                out <- n
            }
        }
    }()
    return out
}

func main() {
    // Pipeline গড়ো: generate → square → filter → print
    nums     := generate(1, 2, 3, 4, 5, 6)
    squared  := square(nums)
    filtered := filter(squared, 10)

    for result := range filtered {
        fmt.Println(result)
    }
}

16   (4²)

Fan-Out / Fan-In প্যাটার্ন উন্নত

Fan-Out: একটি Channel থেকে একাধিক Worker-এ কাজ বিতরণ করা।
Fan-In: একাধিক Channel-এর ডেটা একটিতে জড়ো করা।

go fanout-fanin.go 
package main

import (
    "fmt"
    "sync"
    "time"
    "math/rand"
)

// একটি worker — ইনপুট নিয়ে প্রসেস করে আউটপুট দেয়
func worker(id int, in <-chan int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for n := range in {
            // কাজের সময় অনুকরণ
            time.Sleep(time.Duration(rand.Intn(100)) * time.Millisecond)
            result := n * n
            fmt.Printf("Worker %d: %d² = %d\n", id, n, result)
            out <- result
        }
    }()
    return out
}

// Fan-In: একাধিক channel মিলিয়ে একটি বানাও
func merge(channels ...<-chan int) <-chan int {
    merged := make(chan int)
    var wg sync.WaitGroup

    output := func(c <-chan int) {
        defer wg.Done()
        for v := range c {
            merged <- v
        }
    }

    wg.Add(len(channels))
    for _, c := range channels {
        go output(c)
    }

    // সব শেষ হলে merged বন্ধ করো
    go func() {
        wg.Wait()
        close(merged)
    }()

    return merged
}

func main() {
    input := generate(1, 2, 3, 4, 5, 6)

    // Fan-Out: একই input channel থেকে ৩টি worker-এ বিতরণ
    // (এখানে সবাই একই ch পড়ে — Go নিজেই বিতরণ করে)
    w1 := worker(1, input)
    w2 := worker(2, input)
    w3 := worker(3, input)

    // Fan-In: তিনটি output একসাথে
    for result := range merge(w1, w2, w3) {
        fmt.Println("→ ফলাফল:", result)
    }
}

func generate(nums ...int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for _, n := range nums { out <- n }
    }()
    return out
}

Worker Pool প্যাটার্ন উন্নত

নির্দিষ্ট সংখ্যক Worker Goroutine তৈরি করে একটি job queue থেকে কাজ নেয়। এটি সবচেয়ে বেশি ব্যবহৃত concurrency প্যাটার্নগুলোর একটি।

go worker-pool.go 
package main

import (
    "fmt"
    "sync"
    "time"
)

type Job struct {
    ID    int
    Value int
}

type Result struct {
    Job    Job
    Output int
}

// worker: jobs থেকে কাজ নেয়, results-এ পাঠায়
func workerFn(id int, jobs <-chan Job, results chan<- Result, wg *sync.WaitGroup) {
    defer wg.Done()
    for job := range jobs {
        fmt.Printf("[Worker %d] কাজ #%d শুরু\n", id, job.ID)
        time.Sleep(500 * time.Millisecond)  // কাজের অনুকরণ
        output := job.Value * job.Value
        results <- Result{Job: job, Output: output}
        fmt.Printf("[Worker %d] কাজ #%d শেষ → %d\n", id, job.ID, output)
    }
}

func main() {
    const (
        numJobs    = 10
        numWorkers = 3  // একসাথে সর্বোচ্চ ৩ জন কাজ করবে
    )

    jobs    := make(chan Job, numJobs)
    results := make(chan Result, numJobs)
    var wg sync.WaitGroup

    // ৩টি worker goroutine শুরু করো
    for w := 1; w <= numWorkers; w++ {
        wg.Add(1)
        go workerFn(w, jobs, results, &wg)
    }

    // সব job পাঠাও
    for j := 1; j <= numJobs; j++ {
        jobs <- Job{ID: j, Value: j}
    }
    close(jobs)  // সব কাজ পাঠানো হয়েছে

    // সব worker শেষ হলে results বন্ধ করো
    go func() {
        wg.Wait()
        close(results)
    }()

    // ফলাফল সংগ্রহ করো
    var total int
    for r := range results {
        total += r.Output
    }
    fmt.Println("মোট ফলাফল:", total)
}

Done Channel প্যাটার্ন (Cancellation)

done channel দিয়ে Goroutine-কে বন্ধ করার সংকেত দেওয়া হয়। এটি Go-তে Cancellation-এর মৌলিক পদ্ধতি।

go done-channel.go 
package main

import (
    "fmt"
    "time"
)

// Goroutine যতক্ষণ কাজ করবে, done না পেলে
func worker(done <-chan struct{}) {
    for {
        select {
        case <-done:
            fmt.Println("[Worker] বন্ধ সংকেত পেয়েছি, বিদায়!")
            return
        default:
            fmt.Println("[Worker] কাজ করছি...")
            time.Sleep(500 * time.Millisecond)
        }
    }
}

func main() {
    done := make(chan struct{})

    go worker(done)

    time.Sleep(2 * time.Second)
    close(done)  // সব listener-কে একসাথে সংকেত দাও

    time.Sleep(100 * time.Millisecond)
    fmt.Println("[Main] সম্পন্ন")
}

Channel-এ close() করলে সেই channel-এ অপেক্ষারত সব Goroutine-কে একসাথে জানানো যায়। এটি done <- struct{}{} পাঠানোর চেয়ে ভালো, কারণ শুধু একটি Goroutine সংকেত পেত।

Nil Channel-এর কৌশল

Nil channel-এ send বা receive সবসময় block করে — অনির্দিষ্টকাল। এটি Select-এ একটি case নিষ্ক্রিয় করার কৌশলে ব্যবহৃত হয়।

go nil-channel-trick.go 
package main

import "fmt"

func main() {
    ch1 := make(chan int, 3)
    ch2 := make(chan int, 3)

    ch1 <- 1; ch1 <- 2; ch1 <- 3; close(ch1)
    ch2 <- 10; ch2 <- 20; close(ch2)

    // উভয় channel থেকে পড়তে থাকো, যেটি শেষ হয়ে যায়, সেটি nil করো
    for ch1 != nil || ch2 != nil {
        select {
        case v, ok := <-ch1:
            if !ok {
                ch1 = nil  // এই case কে নিষ্ক্রিয় করো
                fmt.Println("ch1 শেষ")
            } else {
                fmt.Println("ch1:", v)
            }
        case v, ok := <-ch2:
            if !ok {
                ch2 = nil  // এই case কে নিষ্ক্রিয় করো
                fmt.Println("ch2 শেষ")
            } else {
                fmt.Println("ch2:", v)
            }
        }
    }
    fmt.Println("সবকিছু শেষ!")
}

ভুল ২: Goroutine Leak

go ❌ Goroutine Leak 
func leak() <-chan int {
    ch := make(chan int)
    go func() {
        // ❌ যদি কেউ ch না পড়ে, এই goroutine চিরকাল থাকবে
        for v := 0; ; v++ {
            ch <- v
        }
    }()
    return ch
}

// ✅ সমাধান: done channel দিয়ে বন্ধ করার সুযোগ দাও
func noLeak(done <-chan struct{}) <-chan int {
    ch := make(chan int)
    go func() {
        defer close(ch)
        for v := 0; ; v++ {
            select {
            case ch <- v:
            case <-done:
                return  // পরিষ্কারভাবে বের হও
            }
        }
    }()
    return ch
}

বাস্তব জীবনের প্যাটার্ন

Semaphore — একসাথে কতজন কাজ করতে পারবে সীমা দাও

go semaphore.go 
package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    // একসাথে সর্বোচ্চ ৩টি goroutine চলবে
    semaphore := make(chan struct{}, 3)
    var wg sync.WaitGroup

    for i := 1; i <= 10; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()

            // "টিকেট" নাও — slot খালি না হলে অপেক্ষা করো
            semaphore <- struct{}{}
            defer func() { <-semaphore }()  // "টিকেট" ফেরত দাও

            fmt.Printf("[%d] কাজ করছি\n", id)
            time.Sleep(500 * time.Millisecond)
            fmt.Printf("[%d] শেষ\n", id)
        }(i)
    }
    wg.Wait()
}

Future / Promise প্যাটার্ন — পরে ফলাফল নাও

go future-pattern.go 
package main

import (
    "fmt"
    "time"
)

type Future[T any] struct {
    ch <-chan T
}

func (f Future[T]) Get() T {
    return <-f.ch  // ফলাফল না আসা পর্যন্ত অপেক্ষা করো
}

func Async[T any](fn func() T) Future[T] {
    ch := make(chan T, 1)
    go func() { ch <- fn() }()
    return Future[T]{ch: ch}
}

func main() {
    // ব্যাকগ্রাউন্ডে কাজ শুরু করো
    result := Async(func() int {
        time.Sleep(2 * time.Second)
        return 42
    })

    fmt.Println("অন্য কাজ করছি...")
    time.Sleep(500 * time.Millisecond)
    fmt.Println("আরো কাজ...")

    // এখন ফলাফল দরকার — অপেক্ষা করো
    fmt.Println("ফলাফল:", result.Get())
}
